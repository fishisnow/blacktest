# 数据源配置示例
# 复制此文件为 .env 并填入您的配置

# ===== Tushare 配置 =====
# Tushare API Token (从 https://tushare.pro/ 获取)
TUSHARE_TOKEN=your_tushare_token_here
# Tushare 优先级 (数字越小优先级越高)
TUSHARE_PRIORITY=2
# Tushare 超时时间 (秒)
TUSHARE_TIMEOUT=30
# Tushare 重试次数
TUSHARE_RETRY_COUNT=3

# ===== Futu OpenAPI 配置 =====
# 是否启用 Futu 数据源
FUTU_ENABLED=true
# Futu OpenD 连接地址
FUTU_HOST=127.0.0.1
# Futu OpenD 连接端口
FUTU_PORT=11111
# Futu 优先级 (数字越小优先级越高)
FUTU_PRIORITY=1
# Futu 超时时间 (秒)
FUTU_TIMEOUT=30
# Futu 重试次数
FUTU_RETRY_COUNT=3
# Futu 交易时段 (ALL, REGULAR, AFTER_HOURS)
FUTU_SESSION=ALL

# ===== 数据源优先级说明 =====
# 优先级数字越小，优先级越高
# 系统会优先使用优先级高的数据源获取数据
# 如果主要数据源失败，会自动尝试备用数据源
# 
# 推荐配置：
# - 如果主要交易美股/港股，设置 FUTU_PRIORITY=1, TUSHARE_PRIORITY=2
# - 如果主要交易A股，设置 TUSHARE_PRIORITY=1, FUTU_PRIORITY=2
